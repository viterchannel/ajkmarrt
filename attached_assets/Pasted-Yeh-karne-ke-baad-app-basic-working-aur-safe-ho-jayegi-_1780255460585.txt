export function checkFeatureGate(feature: string, user: Rider): boolean {
  if (!user) return false;
  
  switch (feature) {
    case 'accept_ride':
      return (
        user.phoneVerified === true &&
        user.emailVerified === true && // email bhi verified ho
        user.approvalStatus === 'approved' && // admin ne approve kiya ho
        (user.walletBalance ?? 0) >= (user.platformMinBalance ?? 50) // dynamic min balance
      );
    case 'wallet_withdraw':
      return user.emailVerified && user.approvalStatus === 'approved';
    // aur features add kar sakte hain
    default:
      return false; // unknown feature ko false
  }
}

// api-server/src/middleware/featureGate.ts
import { db } from '@ajkmart/db';

export const requireFeatureGate = (feature: string) => {
  return async (req, res, next) => {
    const riderId = req.user?.id; // auth middleware se
    const rider = await db.rider.findUnique({ where: { id: riderId } });
    
    if (!rider) return res.status(403).json({ code: 'RIDER_NOT_FOUND' });

    const allowed = checkFeatureGateForBackend(feature, rider);
    if (!allowed) {
      return res.status(403).json({ 
        code: 'FEATURE_NOT_ALLOWED',
        reason: getBlockReason(feature, rider) // e.g., "Phone not verified"
      });
    }
    next();
  };
};

typescript
async flush() {
  const items = await this.getAll();
  const failedItems = [];
  for (const item of items) {
    try {
      const response = await this.processItem(item);
      if (response.ok || response.status < 400) {
        await this.removeItem(item.id);
      } else {
        failedItems.push({ item, status: response.status });
        // maybe retry count badhao, if max retries reached, mark as failed
      }
    } catch (networkError) {
      failedItems.push({ item, error: 'NETWORK_ERROR' });
    }
  }
  if (failedItems.length > 0) {
    reportErrorToAdmin('OFFLINE_QUEUE_FAILURES', failedItems); // new
  }
}

// apps/rider-app/src/lib/error-reporter.ts
import { apiClient } from './api';

export async function reportError(error: Error, context?: any) {
  // Local log (existing)
  console.error(error, context);
  
  // Send to admin server
  try {
    await apiClient.post('/api/admin/error-reports', {
      message: error.message,
      stack: error.stack,
      context: JSON.stringify(context),
      timestamp: new Date().toISOString(),
      userId: getCurrentUserId(),
    });
  } catch (e) {
    // agar ye bhi fail ho to silently ignore, but log locally
  }
}
Backend endpoint: POST /api/admin/error-reports (admin dashboard mein list view banao).

// Pakistan boundary (rough)
const PAKISTAN_BOUNDS = {
  north: 37.0841,
  south: 23.7695,
  east: 77.8375,
  west: 60.8722,
};

export function isValidCoordinate(lat: number, lng: number): boolean {
  // basic range
  if (lat < -90 || lat > 90 || lng < -180 || lng > 180) return false;
  // reject 0,0
  if (lat === 0 && lng === 0) return false;
  // geofence check
  if (lat < PAKISTAN_BOUNDS.south || lat > PAKISTAN_BOUNDS.north) return false;
  if (lng < PAKISTAN_BOUNDS.west || lng > PAKISTAN_BOUNDS.east) return false;
  return true;
}
